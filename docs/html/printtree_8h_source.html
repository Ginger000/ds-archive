35 template<class NodeDescriptor>
36 void printTree(NodeDescriptor root, std::ostream &out = std::cout) {
37  using namespace private_print_functions;
38 
39  // Base case
40  if (root.isNull()) {
41  out << "(empty)" << std::endl;
42  return;
43  }
44 
45  // Make a character matrix for outputting the tree onto
46  int root_height = getPrintHeight(root);
47  int print_matrix_width = (4 << root_height) - 3;
48  int print_matrix_height = 2 * root_height + 1;
49 
50  // Initialize this matrix to be filled with spaces
51  std::vector<std::string> output(print_matrix_height);
52  for (size_t i = 0; i < output.size(); ++i)
53  output[i] = std::string(print_matrix_width + 4, ' '); // + 4 extra room for long things
54 
55  // Recursively print each node
56  printSubtree(root, output, 0, 0, print_matrix_width);
57 
58  // Output the matrix
59  for (int row = 0; row < print_matrix_height; ++row)
60  out << output[row] << std::endl;
61 }

73 template<class NodeDescriptor>
74 void printSubtree(NodeDescriptor croot, std::vector<std::string> &output, int left, int top, int curr_width) {
75  // Convert data to std::string
76  std::string nodeStr = croot.key();
77 
78  // Output data
79  int left_start_shift = 1 - (nodeStr.length() - 1) / 2;
80  for (size_t i = 0; i < nodeStr.length() && left + curr_width / 2 + i < output[top].length(); i++)
81  output[top][left + curr_width / 2 + left_start_shift + i] = nodeStr[i];
82 
83  // Calculate / \ offset = 2 ^ height
84  int branchOffset = (curr_width + 3) >> 3; //(1 << (node->printData - 1));
85 
86  // Print left child
87  int center = left + curr_width / 2;
88  int leftcenter = left + (curr_width / 2 - 1) / 2;
89  int rightcenter = left + curr_width / 2 + 2 + (curr_width / 2 - 1) / 2;
90 
91  if (!croot.left().isNull()) {
92  int branch_pos = center - branchOffset + 1;
93  // draw left upper branch
94  for (int pos = center + left_start_shift - 2; pos > branch_pos; pos--)
95  output[top][pos] = '_';
96  // draw left '/'
97  output[top + 1][branch_pos] = '/';
98  // draw left lower branch
99  for (int pos = branch_pos - 1; pos > leftcenter + 2; pos--)
100  output[top + 1][pos] = '_';
101  // draw left subtree
102  printSubtree(croot.left(), output, left, top + 2, curr_width / 2 - 1);
103  }
104 
105  // Print right child
106  if (!croot.right().isNull()) {
107  int branch_pos = center + branchOffset + 1;
108  // draw right upper branch
109  for (int pos = center + left_start_shift + nodeStr.length() + 1; pos < branch_pos; pos++)
110  output[top][pos] = '_';
111  // draw right '\'
112  output[top + 1][branch_pos] = '\\';
113  // draw right lower branch
114  for (int pos = branch_pos + 1; pos < rightcenter; pos++)
115  output[top + 1][pos] = '_';
116  // draw right subtree
117  printSubtree(croot.right(), output, left + curr_width / 2 + 2, top + 2, curr_width / 2 - 1);
118  }
